function createTimeDrivenTrigger() {

  // Delete all existing triggers

  var triggers = ScriptApp.getProjectTriggers();

  for (var i = 0; i < triggers.length; i++) {

    ScriptApp.deleteTrigger(triggers[i]);

  }

  // Create a new trigger that runs checkForNewRows every 2 minutes

  ScriptApp.newTrigger('checkForNewRows')

    .timeBased()

    .everyMinutes(5)

    .create();

}

function checkForNewRows() {

  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

  // Use Script Properties to keep track of the last row that was processed

  var scriptProperties = PropertiesService.getScriptProperties();

  var lastRow = scriptProperties.getProperty('lastRow');

  if (lastRow == null) {

    // This is the first time the script is running, so set lastRow to the current number of rows

    lastRow = sheet.getLastRow();

    scriptProperties.setProperty('lastRow', lastRow);

  }

  var currentRow = sheet.getLastRow();

  Logger.log('Last row: ' + lastRow);

  Logger.log('Current row: ' + currentRow);

  if (currentRow > lastRow) {

    // There are new rows, so update the last row in Script Properties

    scriptProperties.setProperty('lastRow', currentRow);

    // Now generate the PDF and send the email for each new row

    for (var row = Number(lastRow) + 1; row <= currentRow; row++) {

      Logger.log('Processing row: ' + row);

      generatePDF(row);

    }

  } else {

    Logger.log('No new rows to process.');

  }

}

function generatePDF(row) {

  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

  var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];

  var rowData = sheet.getRange(row, 1, 1, sheet.getLastColumn()).getValues()[0];

  // Create a temporary Document to generate the PDF

  var doc = DocumentApp.create('Temp Document');

  var body = doc.getBody();

  // Add the data to the document with labels

  for(var i = 0; i < rowData.length; i++) {

    var paragraph = body.appendParagraph(String(headers[i]) + ": " + String(rowData[i]));

    paragraph.editAsText().setBold(0, String(headers[i]).length + 1, true);

  }

  // Save and close the temporary document

  doc.saveAndClose();

  // Convert temporary document to PDF

  var pdf = DriveApp.getFileById(doc.getId()).getAs('application/pdf');

  // Delete the temporary document

  DriveApp.getFileById(doc.getId()).setTrashed(true);

  // Get the recipient's name and email from the row data

  var recipientName = rowData[2]; // replace 2 with the index of the column with the name.

  var recipientEmail = rowData[4]; // replace 4 with the index of the column with the email.

  // Send the PDF by email

  MailApp.sendEmail({

    to: recipientEmail,

    subject: "Here is your PDF reciept, " + recipientName,

    body: "Hi" + recipientName + "!\n\nA PDF receipt of your form submission is attached. Reply directly to this email if you notice any errors.",

    attachments: [pdf]

  });

}
